package com.bdl.config;

import com.bdl.config.ConfigDescription;
import com.bdl.config.ConfigSupplier;
import com.bdl.config.ConfigValue;

import dagger.Module;
import dagger.Provides;
import dagger.multibindings.IntoSet;

/** Dagger module for binding configs in the com.bdl.config package. */
@Module(includes = {com.bdl.config.things.ConfigDaggerModule, com.bdl.config.others.ConfigDaggerModule})
public class ConfigDaggerModule {

  /** Adds a ConfigSupplier for config otherFlag1 (com.bdl.config.others.OtherThingA.otherFlag1) to the set multibinder. */
  @Provides
  @IntoSet
  public static ConfigSupplier provideConfigSupplier_otherFlag1() {
    ConfigDescription description = ConfigDescription.builder()
        .className("OtherThingA")
        .fieldName("otherFlag1")
        .type("java.lang.String")
        .build();
    return ConfigSupplier.simple(description, com.bdl.config.others.OtherThingA.otherFlag1);
  }

  /** Binds the type of the config with a ConfigValue annotation to the Configurable's value. */
  @Provides
  @ConfigValue("otherFlag1")
  public static java.lang.String provideConfigValue_otherFlag1(Configuration configuration) {
    return configuration.get("com.bdl.config.others.OtherThingA.otherFlag1");
  }

  /** Adds a ConfigSupplier for config otherFlag2 (com.bdl.config.others.OtherThingA.otherFlag2) to the set multibinder. */
  @Provides
  @IntoSet
  public static ConfigSupplier provideConfigSupplier_otherFlag2() {
    ConfigDescription description = ConfigDescription.builder()
        .className("OtherThingA")
        .fieldName("otherFlag2")
        .type("java.lang.String")
        .build();
    return ConfigSupplier.reflective(description);
  }

  /** Binds the type of the config with a ConfigValue annotation to the Configurable's value. */
  @Provides
  @ConfigValue("otherFlag2")
  public static java.lang.String provideConfigValue_otherFlag2(Configuration configuration) {
    return configuration.get("com.bdl.config.others.OtherThingA.otherFlag2");
  }

  /** Adds a ConfigSupplier for config flag2 (com.bdl.config.things.Thing2.flag2) to the set multibinder. */
  @Provides
  @IntoSet
  public static ConfigSupplier provideConfigSupplier_flag2() {
    ConfigDescription description = ConfigDescription.builder()
        .className("Thing2")
        .fieldName("flag2")
        .type("java.lang.String")
        .build();
    return ConfigSupplier.simple(description, com.bdl.config.things.Thing2.flag2);
  }

  /** Binds the type of the config with a ConfigValue annotation to the Configurable's value. */
  @Provides
  @com.bdl.config.annotation.processor.Annotations.DummyQualifier("flag2")
  public static java.lang.String provideConfigValue_flag2(Configuration configuration) {
    return configuration.get("com.bdl.config.things.Thing2.flag2");
  }
}
